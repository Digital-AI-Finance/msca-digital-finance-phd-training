% --- Course Card ---
\newcommand{\coursecard}[5]{%
  % #1=Code, #2=Title, #3=ECTS, #4=Provider, #5=Description
  \begin{tcolorbox}[
    colback=mscaLightBlue,
    colframe=mscaMediumBlue,
    title={\sffamily\bfseries #1: #2},
    fonttitle=\color{white},
    coltitle=white,
    colbacktitle=mscaMediumBlue,
    boxrule=0.5pt,
    arc=2mm,
    left=4mm, right=4mm, top=2mm, bottom=2mm
  ]
  \textbf{ECTS:} #3 \hfill \textbf{Provider:} #4 \\[4pt]
  #5
  \end{tcolorbox}
}

% --- Event Page ---
\newcommand{\eventpage}[7]{%
  % #1=Title, #2=Date, #3=Location, #4=Host, #5=Description, #6=Program, #7=Photo path
  \begin{center}
    {\Huge\sffamily\bfseries\color{mscaDarkBlue} #1}\\[8pt]
    {\Large\sffamily\color{mscaMediumBlue} #2 \textbar{} #3}\\[4pt]
    {\large\sffamily Hosted by #4}
  \end{center}
  \vspace{8pt}
  \ifblank{#7}{}{%
    \begin{figure}[h]\centering
      \includegraphics[width=0.85\textwidth]{#7}
    \end{figure}
  }%
  \subsection*{Overview}
  #5
  \subsection*{Program}
  #6
}
